<?php
function getFilesSize($path = '')
{
    $root = $_SERVER['DOCUMENT_ROOT'];
    $fullPath = $root . $path;
    if (strlen($fullPath) > 0) {
        $fileSize = 0;
        $dir = scandir($root);
        foreach ($dir as $file) {
            var_dump($file);
            $fileSize += filesize($file);
        }
        return "Путь к папке (${fullPath}), размер файлов в байтах:${fileSize}";
//    return "Путь к папке " . $root . ", размер файлов в байтах: " . $fileSize;
    }
}
